\def\h{1}
\def\s{0.05}
\def\emptyblock#1#2{
	\fill[color=gray]
		({#1*\h+\s},{-#2*\h+\s})
		rectangle
		({(#1+1)*\h-\s},{(-#2+1)*\h-\s});
	\fill[color=black]
		({#1*\h+2*\s},{-#2*\h+2*\s})
		rectangle
		({(#1+1)*\h-2*\s},{(-#2+1)*\h-2*\s});
}
\def\block#1#2#3#4{
	\fill[color=#4]
		({#1*\h+\s},{-#2*\h+\s})
		rectangle
		({(#1+1)*\h-\s},{(-#2+1)*\h-\s});
	\node[color=white] at ({(#1+0.5)*\h},{(-#2+0.5)*\h-0.04})
		{\Large\texttt{#3}\strut};
}
